# D1 建表语句（SQLite）

## 创建数据库

```bash
wrangler d1 create japanesename-data
```

## 建表 SQL

```sql
-- ============================================================
-- Tag 枚举表
-- 存储所有维度及其枚举值，LLM prompt 和校验时引用
-- ============================================================
CREATE TABLE IF NOT EXISTS tag_dimensions (
  id         INTEGER PRIMARY KEY AUTOINCREMENT,
  dimension  TEXT NOT NULL,                -- gender, era, vibe, use_case...
  value      TEXT NOT NULL,                -- male, female, cute, cool...
  label_en   TEXT,
  label_zh   TEXT,
  sort_order INTEGER DEFAULT 0,
  created_at TEXT DEFAULT (datetime('now')),
  UNIQUE(dimension, value)
);

-- ============================================================
-- Name 库（核心表）
-- 硬属性 = 数据集自带，NOT NULL
-- 软属性 = LLM 打标签，可空（按 status 逐步填充）
-- 多选字段以 JSON 数组存储，用 json_each() 查询
-- ============================================================
CREATE TABLE IF NOT EXISTS names (
  -- 主键 & 基础信息（硬属性，NOT NULL）
  id               TEXT PRIMARY KEY,       -- gn_yui_7d93 / fn_tanaka_a1b2
  romaji           TEXT NOT NULL,          -- Yui
  kanji            TEXT NOT NULL,          -- 結衣
  reading          TEXT NOT NULL,          -- ゆい（平假名）

  gender           TEXT NOT NULL CHECK (gender IN ('female', 'male', 'unisex')),
  name_part        TEXT NOT NULL CHECK (name_part IN ('given_name', 'family_name')),
  syllable_count   INTEGER NOT NULL,       -- 英语音节数（从 romaji 计算）
  mora_count       INTEGER,                -- 日本語モーラ数（从假名 reading 计算，如 きん=2）
  kanji_count      INTEGER,                -- 汉字文字数（如 新垣=2，さくら=0）
  script           TEXT NOT NULL DEFAULT '[]',  -- JSON: ["kanji", "hiragana"]

  -- 辅助查询字段（自动计算）
  romaji_initial   TEXT GENERATED ALWAYS AS (UPPER(SUBSTR(romaji, 1, 1))) STORED,

  -- 软属性（LLM 打标签，可空）
  era              TEXT CHECK (era IN ('ancient', 'traditional', 'modern', '2000s', 'trending')),
  popularity       TEXT CHECK (popularity IN ('very_common', 'common', 'uncommon', 'rare', 'unique')),
  origin_region    TEXT DEFAULT 'japan_native'
                        CHECK (origin_region IN ('japan_native', 'japanese_american', 'international')),
  regional_origin  TEXT,                   -- 地域来源：okinawa, kanto, kansai…（主要用于 family_name）
  use_case         TEXT NOT NULL DEFAULT '[]',  -- JSON: ["real_person", "anime", "baby"]
  vibe             TEXT NOT NULL DEFAULT '[]',  -- JSON: ["cute", "gentle", "elegant"]  ≤3
  element          TEXT DEFAULT '[]',           -- JSON: ["fire", "moon"]  ≤3
  kanji_meaning_tags TEXT NOT NULL DEFAULT '[]', -- JSON: 不限枚举，LLM 尽量多打（10-20个）

  -- 数值数据（可从 myoji-yurai.net 等网站抓取，主要用于 family_name）
  household_count      INTEGER,            -- 世帯数（如 新垣≈200）
  estimated_population INTEGER,            -- 推定人口数（如 新垣≈34000）
  national_rank        INTEGER,            -- 全国姓氏排名（如 新垣=563）

  -- 结构化数据（LLM + 网站数据填充）
  kanji_breakdown       TEXT DEFAULT '[]', -- JSON: [{"kanji":"新","meanings_en":["new","fresh","modern"],"reading":"あら"}]
  alternative_readings  TEXT DEFAULT '[]', -- JSON: ["あらかき","しんがき","にいがき"] 同汉字异读
  reading_romaji_variants TEXT DEFAULT '[]', -- JSON: ["Aragaki","Arakaki"] 罗马字拼写变体
  related_names         TEXT DEFAULT '[]', -- JSON: ["fn_arai_xxxx","fn_araki_xxxx"] 相似名字 ID

  -- 文案（LLM 生成）
  meaning_en       TEXT,                   -- 一句话含义（英）
  meaning_zh       TEXT,                   -- 一句话含义（中）
  description_en   TEXT,                   -- 2-3 句文化解释（英）
  description_zh   TEXT,                   -- 2-3 句文化解释（中）
  etymology_en     TEXT,                   -- 名字的历史/地理起源（如"源自琉球仲城間切新垣村"）
  famous_bearers   TEXT DEFAULT '[]',      -- JSON: [{"name":"Aragaki Yui","context":"actress, born 1988"}]
  kamon_url        TEXT,                   -- 家紋图片 URL（仅 family_name）

  -- 管理
  status           TEXT NOT NULL DEFAULT 'raw'
                        CHECK (status IN ('raw', 'llm_enriched', 'reviewed', 'complete')),
  source           TEXT,                   -- 数据来源标记（如 "jmnedict", "web_scrape", "manual"）
  created_at       TEXT DEFAULT (datetime('now')),
  updated_at       TEXT DEFAULT (datetime('now')),

  -- 唯一约束：同一汉字 + 读音 + 类型 = 同一个名字
  UNIQUE(kanji, reading, name_part)
);

-- 硬属性索引（等值查询）
CREATE INDEX IF NOT EXISTS idx_names_gender ON names(gender);
CREATE INDEX IF NOT EXISTS idx_names_name_part ON names(name_part);
CREATE INDEX IF NOT EXISTS idx_names_popularity ON names(popularity);
CREATE INDEX IF NOT EXISTS idx_names_era ON names(era);
CREATE INDEX IF NOT EXISTS idx_names_syllable ON names(syllable_count);
CREATE INDEX IF NOT EXISTS idx_names_mora ON names(mora_count);
CREATE INDEX IF NOT EXISTS idx_names_kanji_count ON names(kanji_count);
CREATE INDEX IF NOT EXISTS idx_names_status ON names(status);
CREATE INDEX IF NOT EXISTS idx_names_romaji ON names(romaji);
CREATE INDEX IF NOT EXISTS idx_names_romaji_initial ON names(romaji_initial);
CREATE INDEX IF NOT EXISTS idx_names_national_rank ON names(national_rank);

-- 复合索引（最常用查询组合）
CREATE INDEX IF NOT EXISTS idx_names_gender_part ON names(gender, name_part);
CREATE INDEX IF NOT EXISTS idx_names_part_pop ON names(name_part, popularity);
CREATE INDEX IF NOT EXISTS idx_names_part_rank ON names(name_part, national_rank);  -- Top N 姓氏查询

-- ============================================================
-- Keyword 库
-- ============================================================
CREATE TABLE IF NOT EXISTS keywords (
  id                  TEXT PRIMARY KEY,     -- kw_0001
  keyword             TEXT NOT NULL,
  search_volume       INTEGER,
  search_volume_total INTEGER,
  kd                  INTEGER,
  cpc                 REAL,
  intent              TEXT,                 -- I, C, T, N

  strategy            TEXT CHECK (strategy IN ('category_page', 'blog_post', 'tool_page', 'homepage_seo')),
  path                TEXT,                 -- /names/female, /blog/ren-meaning
  priority            INTEGER DEFAULT 5 CHECK (priority BETWEEN 1 AND 5),
  status              TEXT DEFAULT 'draft'
                           CHECK (status IN ('draft', 'seo_ready', 'filter_ready', 'quiz_ready', 'published')),

  -- JSON 字段
  keyword_aliases     TEXT DEFAULT '[]',    -- [{"keyword":"...","search_volume":123}]
  tags_hint           TEXT,                 -- {"gender":"female","vibe":"cute"}
  dimensions          TEXT,                 -- 维度分类
  seo                 TEXT,                 -- {title, description, h1, h2[], keywords[]}
  seo_guidance        TEXT,
  filter_rule         TEXT,                 -- {must:[{field,op,value}], should:[...]}
  quiz                TEXT,                 -- [{question_en, dimension, options[]}]
  related_keywords    TEXT DEFAULT '[]',    -- [{label, path, filter_rules}]

  page_template_type  TEXT,
  relevance_score     INTEGER,
  source              TEXT,                 -- 数据来源标记

  created_at          TEXT DEFAULT (datetime('now')),
  updated_at          TEXT DEFAULT (datetime('now'))
);

CREATE INDEX IF NOT EXISTS idx_keywords_strategy ON keywords(strategy);
CREATE INDEX IF NOT EXISTS idx_keywords_path ON keywords(path);
CREATE INDEX IF NOT EXISTS idx_keywords_priority ON keywords(priority);
CREATE INDEX IF NOT EXISTS idx_keywords_status ON keywords(status);

-- ============================================================
-- 自动更新 updated_at（SQLite trigger）
-- ============================================================
CREATE TRIGGER IF NOT EXISTS names_updated_at
  AFTER UPDATE ON names
  FOR EACH ROW
BEGIN
  UPDATE names SET updated_at = datetime('now') WHERE id = NEW.id;
END;

CREATE TRIGGER IF NOT EXISTS keywords_updated_at
  AFTER UPDATE ON keywords
  FOR EACH ROW
BEGIN
  UPDATE keywords SET updated_at = datetime('now') WHERE id = NEW.id;
END;
```

## 多选字段查询示例

D1 (SQLite) 用 `json_each()` 查询 JSON 数组字段，替代 Postgres 的 `@>` / `&&` 数组运算：

```sql
-- any_of: vibe 包含 "cute" 或 "playful"
SELECT n.* FROM names n
WHERE EXISTS (
  SELECT 1 FROM json_each(n.vibe) j
  WHERE j.value IN ('cute', 'playful')
);

-- eq + any_of: gender=female 且 use_case 包含 "anime"
SELECT n.* FROM names n
WHERE n.gender IN ('female', 'unisex')
AND EXISTS (
  SELECT 1 FROM json_each(n.use_case) j
  WHERE j.value = 'anime'
);

-- all_of: element 同时包含 "fire" 和 "dark"
SELECT n.* FROM names n
WHERE (
  SELECT COUNT(DISTINCT j.value) FROM json_each(n.element) j
  WHERE j.value IN ('fire', 'dark')
) = 2;

-- 组合查询: 用于 PSEO 内页 filter_rule 执行
-- 示例: cute female given names, prefer modern era
SELECT n.*,
  CASE WHEN n.era IN ('modern', '2000s') THEN 1 ELSE 0 END AS era_score
FROM names n
WHERE n.gender IN ('female', 'unisex')
  AND n.name_part = 'given_name'
  AND EXISTS (
    SELECT 1 FROM json_each(n.vibe) j
    WHERE j.value IN ('cute', 'playful')
  )
ORDER BY era_score DESC, RANDOM()
LIMIT 200;
```

## 扩展查询示例

新增 DSL 运算符对应的 SQL 写法：

```sql
-- starts_with: 名字以 K 开头
SELECT * FROM names
WHERE romaji_initial = 'K'
ORDER BY popularity DESC, RANDOM()
LIMIT 200;

-- 如果需要精确前缀匹配（多字母）
SELECT * FROM names
WHERE romaji LIKE 'Ka%'
ORDER BY RANDOM()
LIMIT 200;

-- gt / lt / gte / lte: 短名字（音节数 ≤ 2）
SELECT * FROM names
WHERE syllable_count <= 2
ORDER BY RANDOM()
LIMIT 200;

-- kanji_meaning_tags 兜底查询: "names meaning death"
SELECT n.* FROM names n
WHERE EXISTS (
  SELECT 1 FROM json_each(n.kanji_meaning_tags) j
  WHERE j.value IN ('death', 'perish', 'underworld', 'dark', 'shadow')
)
ORDER BY RANDOM()
LIMIT 200;
```

## 覆盖率检查查询

keyword 发布前验证匹配名字数量：

```sql
-- 单个 keyword 覆盖率检查（以 cute female names 为例）
SELECT COUNT(*) AS match_count FROM names n
WHERE n.gender IN ('female', 'unisex')
  AND n.name_part = 'given_name'
  AND EXISTS (
    SELECT 1 FROM json_each(n.vibe) j
    WHERE j.value IN ('cute', 'playful')
  );

-- 判断标准:
-- >= 50    → category_page 正常发布
-- 20~49   → 标记 thin_risk
-- 5~19    → 降级为 blog_post
-- < 5     → 暂不发布
```

## 新字段扩展查询示例

```sql
-- Top 100 日本姓氏
SELECT * FROM names
WHERE name_part = 'family_name' AND national_rank IS NOT NULL
ORDER BY national_rank ASC
LIMIT 100;

-- 冲绳特色姓氏
SELECT * FROM names
WHERE name_part = 'family_name' AND regional_origin = 'okinawa'
ORDER BY national_rank ASC;

-- 单汉字名
SELECT * FROM names
WHERE kanji_count = 1 AND name_part = 'given_name'
ORDER BY RANDOM()
LIMIT 200;

-- 短名字（日语 mora ≤ 2）
SELECT * FROM names
WHERE mora_count <= 2
ORDER BY RANDOM()
LIMIT 200;

-- 人口超过 10 万的常见姓氏
SELECT * FROM names
WHERE name_part = 'family_name' AND estimated_population >= 100000
ORDER BY estimated_population DESC;
```
